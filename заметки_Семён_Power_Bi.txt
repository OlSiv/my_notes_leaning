VAR TotalRevenue = SUM('Продажи'[Выручка])
RETURN 
    ADDCOLUMNS(
        SUMMARIZE(
            'Продажи',
            'Продажи'[Продукт],
            "Выручка по Продукту", SUM('Продажи'[Выручка])
        ),
        "Кумулятивная Выручка", 
        CALCULATE(
            SUM('Продажи'[Выручка]),
            FILTER(
                ALL('Продажи'),
                'Продажи'[Выручка] >= [Выручка по Продукту]
            )
        ),
        "Категория АВС", 
        SWITCH(
            TRUE(),
            CALCULATE(
                SUM('Продажи'[Выручка]),
                FILTER(
                    ALL('Продажи'),
                    'Продажи'[Выручка] >= [Выручка по Продукту]
                )
            ) <= 0.8 * TotalRevenue, "A",
            CALCULATE(
                SUM('Продажи'[Выручка]),
                FILTER(
                    ALL('Продажи'),
                    'Продажи'[Выручка] >= [Выручка по Продукту]
                )
            ) <= 0.95 * TotalRevenue, "B",
            "C"
        )
    )
