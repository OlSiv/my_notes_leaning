RFM_Analysis = 
VAR RFM_Table =
    ADDCOLUMNS(
        SUMMARIZE(
            'Orders', 
            'Orders'[Customer ID],
            "Recency", DATEDIFF( MAX('Orders'[Order Date]), TODAY(), DAY),
            "Frequency", COUNT('Orders'[Order ID]),
            "Monetary", SUM('Orders'[Sales])
        ),
        "Recency Score", 
            SWITCH(TRUE(),
                [Recency] <= 30, 5,
                [Recency] <= 90, 4,
                [Recency] <= 180, 3,
                [Recency] <= 365, 2,
                1
            ),
        "Frequency Score", 
            SWITCH(TRUE(),
                [Frequency] >= 50, 5,
                [Frequency] >= 20, 4,
                [Frequency] >= 10, 3,
                [Frequency] >= 5, 2,
                1
            ),
        "Monetary Score", 
            SWITCH(TRUE(),
                [Monetary] >= 50000, 5,
                [Monetary] >= 20000, 4,
                [Monetary] >= 10000, 3,
                [Monetary] >= 5000, 2,
                1
            )
    )

VAR RFM_Scored_Table = 
    ADDCOLUMNS(
        RFM_Table,
        "RFM Score", 
            FORMAT([Recency Score], "0") & 
            FORMAT([Frequency Score], "0") & 
            FORMAT([Monetary Score], "0")
    )

VAR Aggregated_Table = 
    SUMMARIZE(
        RFM_Scored_Table,
        [RFM Score],
        "Summ Sales", SUM([Monetary]),
        "Count Customer ID", COUNT([Customer ID])
    )

VAR TotalCustomers = CALCULATE(COUNT('Orders'[Customer ID]), ALL('Orders'))

RETURN 
    ADDCOLUMNS(
        Aggregated_Table,
        "% Count Customer ID", 
            FORMAT(DIVIDE([Count Customer ID], TotalCustomers, 0), "0.0%")
    )
