RFM_Analysis = 
VAR RFM_Table =
    ADDCOLUMNS(
        SUMMARIZE(
            'Orders', 
            'Orders'[Customer ID],
            "Recency", DATEDIFF( MAX('Orders'[Order Date]), TODAY(), DAY),
            "Frequency", COUNT('Orders'[Order ID]),
            "Monetary", SUM('Orders'[Sales])
        ),
        "Recency Score", 
            SWITCH(TRUE(),
                [Recency] <= 30, 5,
                [Recency] <= 90, 4,
                [Recency] <= 180, 3,
                [Recency] <= 365, 2,
                1
            ),
        "Frequency Score", 
            SWITCH(TRUE(),
                [Frequency] >= 50, 5,
                [Frequency] >= 20, 4,
                [Frequency] >= 10, 3,
                [Frequency] >= 5, 2,
                1
            ),
        "Monetary Score", 
            SWITCH(TRUE(),
                [Monetary] >= 50000, 5,
                [Monetary] >= 20000, 4,
                [Monetary] >= 10000, 3,
                [Monetary] >= 5000, 2,
                1
            )
    )

VAR RFM_Scored_Table = 
    ADDCOLUMNS(
        RFM_Table,
        "RFM Score", 
            FORMAT([Recency Score], "0") & 
            FORMAT([Frequency Score], "0") & 
            FORMAT([Monetary Score], "0")
    )

VAR Aggregated_Table = 
    SUMMARIZE(
        RFM_Scored_Table,
        [RFM Score],
        "Summ Sales", SUM([Monetary]),
        "Count Customer ID", COUNT([Customer ID])
    )

VAR TotalCustomers = CALCULATE(COUNT('Orders'[Customer ID]), ALL('Orders'))

RETURN 
    ADDCOLUMNS(
        Aggregated_Table,
        "% Count Customer ID", 
            FORMAT(DIVIDE([Count Customer ID], TotalCustomers, 0), "0.0%")
    )

        RFM_Base,

VAR CalendarTable = 
    ADDCOLUMNS (
        CALENDARAUTO(),
        "Год", YEAR([Date]),
        "Начало года", FORMAT(DATE(YEAR([Date]), 1, 1), "dd.mm.yyyy"),
        "Квартал", QUARTER([Date]),
        "Начало квартала", FORMAT(DATE(YEAR([Date]), (QUARTER([Date]) - 1) * 3 + 1, 1), "dd.mm.yyyy"),
        "Месяц", MONTH([Date]),
        "Название месяца", FORMAT([Date], "MMMM"),
        "Неделя года", WEEKNUM([Date], 2),
        "Неделя месяца", IF(WEEKNUM([Date], 2) = WEEKNUM(DATE(YEAR([Date]), MONTH([Date]), 1), 2), 1, WEEKNUM([Date], 2) - WEEKNUM(DATE(YEAR([Date]), MONTH([Date]), 1), 2) + 1),
        "Начало месяца", FORMAT(DATE(YEAR([Date]), MONTH([Date]), 1), "dd.mm.yyyy"),
        "Начало недели", FORMAT([Date] - WEEKDAY([Date], 2) + 1, "dd.mm.yyyy"),
        "День месяца", DAY([Date]),
        "День недели", WEEKDAY([Date], 2),
        "День года", [Date] - DATE(YEAR([Date]), 1, 1) + 1,

        "Название дня (Сокр.)", FORMAT([Date], "ddd"),
        "Год (Сокр.)", RIGHT(YEAR([Date]), 2)
    )

в жпт - какие бизнес-метрики испльзуются в бизнесе по автоперевозкам - некоторые из ключевых метрик -
--
1. **Выручка на грузоперевозку**: Общая сумма доходов от перевозок за определенный период времени.

2. **Себестоимость перевозки**: Все затраты, связанные с выполнением перевозок, включая топливо, зарплату водителей, амортизацию транспортных средств и другие операционные расходы.

3. **Прибыль на перевозку**: Разница между выручкой и себестоимостью перевозки. Позволяет оценить рентабельность каждой перевозки.

4. **Коэффициент загрузки**: Процентное соотношение между фактическим объемом груза и максимальным объемом, который может быть перевезен транспортным средством. Высокий коэффициент загрузки указывает на эффективное использование ресурсов.

5. **Время в пути**: Среднее время, необходимое для выполнения перевозки. Важно для оценки эффективности логистики и планирования маршрутов.

6. **Своевременность доставки**: Процент доставок, выполненных в срок. Важная метрика для оценки качества обслуживания клиентов.

7. **Количество возвратов и рекламаций**: Количество случаев, когда груз был возвращен или возникли жалобы от клиентов. Это может указывать на проблемы с качеством услуг.

8. **Коэффициент удержания клиентов**: Процент клиентов, которые продолжают пользоваться услугами компании в течение определенного периода времени. Высокий коэффициент указывает на удовлетворенность клиентов.

9. **Затраты на топливо**: Средние затраты на топливо на одну перевозку или на определенный период времени. Важная метрика для управления затратами.

10. **Коэффициент аварийности**: Количество аварий на 1000 перевозок или на определенный период времени. Важно для оценки безопасности и надежности автопарка.

11. **Производительность водителей**: Оценка эффективности работы водителей, включая количество выполненных рейсов, среднее время в пути и соблюдение графиков.

12. **Обслуживание и ремонт**: Затраты на техническое обслуживание и ремонт транспортных средств, а также время простоя из-за поломок.

Эти метрики помогают компаниям по автоперевозкам принимать обоснованные решения, оптимизировать процессы и повышать уровень обслуживания клиентов.
